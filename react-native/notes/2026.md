```tsx
import React from 'react';
import { View, Button, ScrollView, Alert } from 'react-native';
import { useForm } from 'react-hook-form';
import { zodResolver } from '@hookform/resolvers/zod';
import * as z from 'zod';
import { FormInput } from '../components/Form/FormInputs';

// 1. Define your validation schema
const registrationSchema = z.object({
  username: z.string().min(3, "Username must be at least 3 characters"),
  email: z.string().email("Invalid email address"),
  phone: z.string().min(10, "Phone number must be at least 10 digits"),
  amount: z.string().refine((val) => !isNaN(Number(val)) && Number(val) > 0, {
    message: "Amount must be a positive number",
  }),
  bio: z.string().max(100, "Bio cannot exceed 100 characters").optional(),
});

// 2. Extract the TypeScript type from the schema
type RegistrationData = z.infer<typeof registrationSchema>;

export default function UserRegistration() {
  // 3. Initialize useForm with the zodResolver
  const { control, handleSubmit, formState: { errors } } = useForm<RegistrationData>({
    resolver: zodResolver(registrationSchema),
    defaultValues: {
      username: '',
      email: '',
      phone: '',
      amount: '',
      bio: '',
    },
  });

  const onSubmit = (data: RegistrationData) => {
    Alert.alert("Success", JSON.stringify(data));
  };

  return (
    <ScrollView className="flex-1 bg-gray-50 p-5">
      {/* Text Input */}
      <FormInput
        name="username"
        control={control}
        label="Username"
        placeholder="e.g. johndoe"
      />

      {/* Email Input */}
      <FormInput
        name="email"
        control={control}
        label="Email Address"
        placeholder="name@example.com"
        inputProps={{ keyboardType: 'email-address', autoCapitalize: 'none' }}
      />

      {/* Phone Number Input */}
      <FormInput
        name="phone"
        control={control}
        label="Phone Number"
        placeholder="+1 234 567 890"
        inputProps={{ keyboardType: 'phone-pad' }}
      />

      {/* Amount Input */}
      <FormInput
        name="amount"
        control={control}
        label="Deposit Amount"
        placeholder="0.00"
        inputProps={{ keyboardType: 'numeric' }}
      />

      {/* Description / Bio Input */}
      <FormInput
        name="bio"
        control={control}
        label="Short Bio"
        placeholder="Tell us about yourself..."
        inputProps={{ multiline: true, numberOfLines: 4, textAlignVertical: 'top' }}
      />

      <View className="mt-4">
        <Button title="Register" onPress={handleSubmit(onSubmit)} color="#007AFF" />
      </View>
    </ScrollView>
  );
}
```

# Linking OPENURL
```tsx
import { Linking, Pressable, Text } from 'react-native';

const openWebsite = async () => {
  const url = 'https://google.com';
  
  // It's best practice to check if the link is valid/supported first
  const supported = await Linking.canOpenURL(url);

  if (supported) {
    await Linking.openURL(url);
  } else {
    console.log("Don't know how to open this URL: " + url);
  }
};

// ... inside your component
<Pressable onPress={openWebsite}>
  <Text>Visit Google</Text>
</Pressable>

```
